/**
 * @function count
 * @description Counts characters in string
 * @param {String} str
 * @return {Number} Number of characters
 * @example
 * count('test')
 * // 4
 * @example
 * str('test').length
 * // 4
 * @example
 * str('test').count()
 * // 4
 */

export default str => String(str).length
