/**
 * @function lower
 * @description Transform str to lower case
 * @param {String} str
 * @return {String} Lower cased string
 * @example
 * lower('TEST')
 * // test
 * @example
 * str('TEST').lower().value
 * // test
 */
export default str => str.toLowerCase()
